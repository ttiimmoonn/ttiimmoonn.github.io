#создание списка
op = ['1', '2', '3', '4', '5', '6', '7']
print("op = ", op)
op1 = list(range(8,13))
print("op1 = ", op1)
op2 = list('Good men')
print("op2 = ", op2)
op3 = ['FastEthernet0/'+ str(i) for i in range(5)]
print("op3 = ", op3)

#переназначить элемент
op[0] = '0'
print("New op = ", op)

#перевернуть список
op.reverse()
print("Invert op = ", op)

#сортировака списка
op.sort()

#объединить в одну строку с указанным разделителем
print("Срока из списка op: ", '+'.join(op))

#разделить строку с указанным разделителем
print("Список из строки op.join", '+'.join(op).split('+'))

#Добавить в концец списка указанный элемент
op.append('1')
print("Cписок op с добавленным элементом: ", op)

#Объеденить два списка через два способа
op.extend(op1)
print("Список op и список op1: ", op)
op = op + op1
print("Сумма op и op1: ",op)

#Удалить элемент списка по номеру (возвращает удаленный элемент)
a = len(op)-1
ab = op.pop(a) 
print("Список op с удаленным элементом", ab, ": ", op)

#Удалить указанный элемент списка(не возвращает удаленный элемент)
op.remove('7') 
print("Список op с удаленным элементом: ", op)

#Найти индекс элемента в списке
print("Элелемент 5 в списке на месте: ", op.index('5'))

#Вставить элемент на определенное место в списке
op.insert(op.index('5'),'6')
print("Список op с добавленным элементом 6 на место 5", op)
